import os
import sys
import django

sys.path.append(os.path.dirname(os.path.abspath(__file__)))
os.environ.setdefault('DJANGO_SETTINGS_MODULE', 'ecole_moderne.settings')
django.setup()

from eleves.models import Eleve
from notes.models import ClasseNote
from notes.export_classement import _generer_classement_general

print("\n" + "="*80)
print("TEST DE COHERENCE - LOUAMMOU Jean David")
print("="*80)

# Trouver l'élève LOUAMMOU Jean David
eleve = Eleve.objects.filter(
    nom__icontains="LOUAMMOU",
    prenom__icontains="Jean",
    statut='ACTIF'
).first()

if not eleve:
    print("\n❌ Élève LOUAMMOU Jean David non trouvé")
    print("Recherche d'une autre classe...")
    eleve = Eleve.objects.filter(statut='ACTIF').first()
    if not eleve:
        print("❌ Aucun élève actif trouvé")
        sys.exit(1)

print(f"\n📋 Élève trouvé : {eleve.nom} {eleve.prenom}")
print(f"   Matricule : {eleve.matricule}")
print(f"   Classe : {eleve.classe.nom}")

# Récupérer tous les élèves de la classe
eleves = Eleve.objects.filter(
    classe=eleve.classe,
    statut='ACTIF'
).order_by('nom', 'prenom')

print(f"   Effectif : {eleves.count()} élèves")

# Trouver la ClasseNote correspondante
classe_note = ClasseNote.objects.filter(
    nom__icontains=eleve.classe.nom.split()[0]
).first()

if not classe_note:
    print(f"\n❌ ClasseNote non trouvée pour {eleve.classe.nom}")
    sys.exit(1)

print(f"   ClasseNote : {classe_note.nom}")

# Générer le classement
print(f"\n📊 Génération du classement OCTOBRE...")
try:
    classement_data, titre = _generer_classement_general(
        eleves, classe_note, 'mensuelle', 'OCTOBRE'
    )
    
    print(f"✅ Classement généré avec succès")
    
    # Trouver la position de l'élève test
    position_eleve = None
    for i, data in enumerate(classement_data, 1):
        if data.get('matricule') == eleve.matricule:
            position_eleve = data
            break
    
    if position_eleve:
        print(f"\n" + "="*80)
        print(f"RÉSULTAT POUR {eleve.nom} {eleve.prenom}")
        print("="*80)
        print(f"Matricule : {position_eleve.get('matricule')}")
        print(f"Rang      : {position_eleve.get('rang')}")
        print(f"Moyenne   : {position_eleve.get('moyenne'):.2f}/20")
        print("="*80)
    
    # Afficher le classement complet
    print(f"\n📋 CLASSEMENT COMPLET - {titre}")
    print("-"*80)
    print(f"{'Rang':<15} {'Matricule':<15} {'Nom Complet':<35} {'Moyenne':<10}")
    print("-"*80)
    
    for data in classement_data:
        if data.get('moyenne') is not None:
            rang = data.get('rang', 'N/A')
            matricule = data.get('matricule', 'N/A')
            nom = data.get('nom_complet', 'N/A')[:34]
            moyenne = data.get('moyenne')
            
            marker = " ⭐" if matricule == eleve.matricule else ""
            print(f"{rang:<15} {matricule:<15} {nom:<35} {moyenne:<10.2f}{marker}")
    
    print("-"*80)
    print(f"\n✅ TEST DE COHÉRENCE : OK")
    print(f"   Total élèves classés : {len([d for d in classement_data if d.get('moyenne') is not None])}")
    
except Exception as e:
    print(f"\n❌ Erreur : {e}")
    import traceback
    traceback.print_exc()
    sys.exit(1)

print("\n" + "="*80)
print("PROCHAINE ÉTAPE")
